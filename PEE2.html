2.1. Entitate baten datuak deskargatu eta irudikatu¶

Notebook honen helburua, B Eredua merkatu errealetako konpainiekin alderatzea da. Horretarako, yahoo finance -tik konpainia bati dagozkion datuak nola deskargatzen eta manipulatzen diren jakin behar dugu. Adibide gisan, IBM konpainia hartu da:

%matplotlib inline

#----------- LIBRERIAK ----------------
       
import numpy as np
import matplotlib as mpl
from matplotlib import rc                  # Latex-en idazteko
import yfinance as yf            # Merkatu errealetako datuak irakurtzeko baliabideak
import matplotlib.pyplot as plt
plt.style.use('fivethirtyeight')

#---------- FUNTZIO BALIAG. ----------
def normalizatu(data):                       # normalizazio hau egiten da bariantza unitarioa lortzeko (hau gabe ere egin daiteke)
    return ((data)/np.sqrt(np.var(data)))

def itzulkin_log(Pt,tau=1):                  # Denbora-serie baten itzulkin logaritmikoa bueltatuko digun funtzioa
    return np.log(Pt[tau:])-np.log(Pt[0:-tau])

def pbanaketa(data,bins=50):
    hist,ertzak=np.histogram(data[~np.isnan(data)],bins=bins,density=True) # NAN-ak kendu eta histograma kalkulatu
    ertzak   = (ertzak[:-1] + ertzak[1:])/2.0 
    ez_zero = hist > 0.0                   
    return  hist[ez_zero],ertzak[ez_zero]

ticker = yf.Ticker('IBM')
datuak = ticker.history(period="max")
logret = normalizatu(itzulkin_log(datuak['Close'].values,1)) 

#----------- DATUEN ESKURATZEA ----------------

# Latex-en idazteko
mpl.rc('xtick', labelsize=20) 
mpl.rc('ytick', labelsize=20)

plt.rc('font', family='serif')

fig, ax = plt.subplots(figsize=(17,6),dpi=300)

ax.plot(datuak['Close'],lw=2)
ax.set_ylabel(r"IBM-en Itxiera-prezioak (USD)",fontsize=20)
ax.set_xlabel(r"Denbora (Urteak)",fontsize=20)

2.1.1 Entitatearen prezioen itzulkin logaritmikoa kalkulatu¶

# Latex-en idazteko
mpl.rc('xtick', labelsize=18) 
mpl.rc('ytick', labelsize=18) 
plt.rc('text', usetex=True)
plt.rc('font', family='serif')

fig, ax= plt.subplots(figsize=(14,7),dpi=200)
hist_konpainia, be_konpainia = pbanaketa(np.abs(logret),bins=100)

x = np.linspace(0.1, 10)

ax.plot(x, 2*np.exp(-1.5*x),lw=4,color='y',ls='--',alpha=0.5,label=r'Esponentziala')
ax.plot(x, 0.3*x**(-3), lw=4, color='g', ls='--', alpha=0.5, label=r'Berretura-legea $\propto x^{-3}$')
ax.plot(be_konpainia, hist_konpainia, lw=1.5, label='IBM')
ax.set_xlabel(r"Prezioen itzulkin logaritmikoak",fontsize=20)
ax.set_ylabel(r"Probabilitate banaketa",fontsize=20)
ax.set_xlim(5e-1, 1e1)
ax.set_ylim(5e-5, 1)
ax.legend(fontsize=18)
ax.semilogx()
ax.semilogy()
plt.show()

2.2. B Ereduaren egiaztapena¶

Notebook honetan, B Eredurako idatziriko koda berretsiko da. Horretarako, merkatu errealetako datuak deskargatuko ditugu, 6. notebookean egin zen bezala. IBMz gain, honengandik desberdintzen den Amazon multinazionalaren itxiera-prezioen itzulkinak ere deskargatuko ditugu (konparaketa orokorragoa izan dadin).

ticker_1 = yf.Ticker('AMZN')
ticker_2 = yf.Ticker('IBM')

df_1= ticker_1.history(period="max")
df_2 = ticker_2.history(period="max")
